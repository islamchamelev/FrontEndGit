266      public static String getProperyFromXml(String xml,String property) throws ParserConfigurationException, SAXException, IOException { 
267          InputSource is = new InputSource(); 
268          is.setCharacterStream(new StringReader(xml)); 
269   
270          DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance(); 
271          DocumentBuilder db = dbf.newDocumentBuilder(); 
272   
273          Document doc = db.parse(is); 
274   
275   
276          NodeList nodes = doc.getElementsByTagName(property); 
277   
278          String thisMessageType = null; 
279          try { 
280              thisMessageType = nodes.item(0).getTextContent(); 
281          } catch (DOMException e) { 
282              e.printStackTrace(); 
283              System.out.println(e); 
284          } 
285   
286          return thisMessageType; 
287      }
